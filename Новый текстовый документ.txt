Cat-Fighting/
│
├── client/ (корень репозитория, деплоится на Vercel)
│   ├── index.html                  # Главная страница, подключает CSS/JS, модальные окна
│   ├── style.css                   # Все стили (адаптивность, анимации, темы, башня, туториал, таймер)
│   ├── app.js                      # Инициализация Telegram WebApp, загрузка данных, рендер экранов (главный, экипировка, профиль, рейтинг)
│   ├── battleUI.js                 # Экран боя: отрисовка, таймер, обработка результата, статистика
│   ├── battleLog.js                # Пошаговое воспроизведение боя: анимации, иконки статусов, всплывающие числа, задержки для берсерка
│   ├── constants.js                # Словари: переводы классов/подклассов, базовые статы, веса силы, переводы предметов, заданий, скинов
│   ├── helpers.js                  # Вспомогательные функции: расчёт статов, силы, русские названия классов, сравнение предметов, модалки
│   ├── forge.js                    # Кузница: слоты, добавление/удаление предметов, ковка, плавка
│   ├── profile.js                  # Профиль: бонусы (статы), улучшение характеристик, скины
│   ├── task-up.js                  # Адвент-календарь, ежедневные задания, реферальная программа, таймер сброса
│   └── tower.js                    # Башня: отображение этажей, туториал, выбор класса на сезон, бои, награды, кнопка помощи
│
├── server/ (деплоится на Render)
│   ├── index.js                    # Точка входа: подключение маршрутов, webhook Telegram, инициализация БД, запуск сервера
│   ├── db.js                       # Пул подключений к PostgreSQL, функция initDB (создание таблиц)
│   ├── package.json                # Зависимости (express, pg, cors, dotenv)
│   │
│   ├── routes/
│   │   ├── auth.js                 # Авторизация через Telegram initData, создание/обновление пользователя, загрузка инвентаря и классов
│   │   ├── battle.js               # Симуляция боя: расчёт урона, эффекты, генерация противника (PvP/бот/киберкот), обновление серии побед
│   │   ├── forge-server.js         # Кузница: добавление/удаление предметов, получение списка, ковка, плавка
│   │   ├── inventory.js            # Надевание/снятие предметов, продажа (выставление на маркет), снятие с продажи
│   │   ├── market.js               # Маркет: получение списка с фильтрами, покупка, изменение цены, снятие с продажи
│   │   ├── player.js               # Смена класса/подкласса, прокачка характеристик, выбор аватара
│   │   ├── rank.js                 # Рейтинги: по очкам, по силе, по башне
│   │   ├── shop.js                 # Покупка сундуков (обычный, необычный, редкий, эпический, легендарный)
│   │   ├── tasks.js                # Ежедневные задания (список, обновление прогресса, получение награды), адвент-календарь (битовая маска)
│   │   ├── tower-server.js         # Башня: прогресс, бои (генерация боссов каждые 5 этажей), награды
│   │   └── avatars.js              # Аватары: список, купленные аватары, покупка
│   │
│   ├── utils/
│   │   ├── botGenerator.js         # Генерация ботов (обычные, киберкоты, боссы-мыши) с очками навыков и статами
│   │   └── power.js                # Расчёт силы персонажа (с учётом навыков, экипировки, подкласса)
│   │
│   └── data/
│       ├── battlePhrases.js        # Тексты атак, критических ударов, уворотов, ультимейтов, эффектов
│       └── itemData.js             # Списки названий предметов и фиксированные бонусы по редкости
│
└── (внешние сервисы)
    ├── GitHub                      # Хранилище кода, интеграция с Vercel и Render
    ├── Vercel                      # Хостинг статических файлов (клиент)
    ├── Render                      # Хостинг Node.js-сервера (бэкенд)
    └── Supabase                    # PostgreSQL база данных (через пулированное соединение)
